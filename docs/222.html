m =folium.Map([48, -102], zoom_start=3)

m.choropleth(geo_data=state_geo, data=state_data, name='us',
            columns =['State','Unemployment'], key_on='feature.id',
            fill_color = 'BuPu', fill_opacity=0.7, line_opacity=0.2,
            legend_name = 'unemployment Rate (%)')
folium.LayerControl().add_to(m)

"""
geo_data : json 파일
data : data 파일
name : 그래프 이름
columns : data의 축 사용
key_on : json 파일의 변수명 지정
fill_color ; 그래프 색
fill_opacity : 채우는 색 투명도
line_opacity : 그래프 구분하는 선 투명도
legend_name : lenged 이름

folium.LayerControl().add_to() 우측 동적 기능 추가
"""
